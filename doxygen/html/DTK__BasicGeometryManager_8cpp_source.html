82  const Teuchos::RCP<const Teuchos::Comm<int> > comm,
83  const int physical_dimension,
84  const EntityType entity_type,
85  const Teuchos::ArrayView<Entity>& entities,
86  const Teuchos::ArrayView<int>& block_ids,

91  BlockPredicate block_pred( block_array );
92  BoundaryPredicate boundary_pred( boundary_array );
93  std::function<bool(Entity)> pred =
94  PredicateComposition::Or( block_pred.getFunction(),
95  boundary_pred.getFunction() );
96  Teuchos::RCP<EntitySelector> entity_selector =
97  Teuchos::rcp( new EntitySelector(entity_type,pred) );
98  createFunctionSpace( comm, physical_dimension, entities, entity_selector );

118  const int physical_dimension,
119  const Teuchos::ArrayView<Entity>& entities,
120  const Teuchos::RCP<EntitySelector>& entity_selector )

129  Teuchos::RCP<EntityLocalMap> local_map =
130  Teuchos::rcp( new BasicGeometryLocalMap() );

133  Teuchos::rcp( new EntityCenteredShapeFunction() );
134
