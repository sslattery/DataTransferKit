86 //---------------------------------------------------------------------------//
87 // Perform a safeguard check for mapping a point to the reference space
88 // of an entity using the given tolerance. Default implementation checks if
89 // the point is inside the bounding box of the entity.
90 bool EntityLocalMap::isSafeToMapToReferenceFrame(
91  const Entity& entity,
92  const Teuchos::ArrayView<const double>& point,
93  const Teuchos::RCP<MappingStatus>& status ) const
94 {
95  // Get the test tolerance.
96  double tolerance = 1.0e-6;
97  if ( Teuchos::nonnull(this->b_parameters) )
98  {
99  if ( b_parameters->isParameter("Point Inclusion Tolerance") )
100  {
101  tolerance = b_parameters->get<double>("Point Inclusion Tolerance");
102  }
103  }
104 
105  // Get the bounding box of the entity.
106  Teuchos::Tuple<double,6> entity_box;
107  entity.boundingBox( entity_box );
108 
109  // Check if the point is in the bounding box of the entity.
110  int space_dim = entity.physicalDimension();
111  bool in_x = true;
112  if ( space_dim > 0 )
113  {
114  double x_tol = (entity_box[3] - entity_box[0])*tolerance;
115  in_x = ( (point[0] >= (entity_box[0] - x_tol)) &&
116  (point[0] <= (entity_box[3] + x_tol)) );
117  }
118  bool in_y = true;
119  if ( space_dim > 1 )
120  {
121  double y_tol = (entity_box[4] - entity_box[1])*tolerance;
122  in_y = ( (point[1] >= (entity_box[1] - y_tol)) &&
123  (point[1] <= (entity_box[4] + y_tol)) );
124  }
125  bool in_z = true;
126  if ( space_dim > 2 )
127  {
128  double z_tol = (entity_box[5] - entity_box[2])*tolerance;
129  in_z = ( (point[2] >= (entity_box[2] - z_tol)) &&
130  (point[2] <= (entity_box[5] + z_tol)) );
131  }
132  return (in_x && in_y && in_z);
133 }

170 //---------------------------------------------------------------------------//
171 // Compute the normal on a face (3D) or edge (2D) at a given reference point.
172 void EntityLocalMap::normalAtReferencePoint(
173  const Entity& entity,
174  const Teuchos::ArrayView<double>& reference_point,
175  const Teuchos::ArrayView<double>& normal ) const
176 {
177  // Determine the reference dimension.
178  int physical_dim = entity.physicalDimension();
179  int ref_dim = physical_dim - 1;
180 
181  // Create a perturbation.
182  double perturbation =
183  std::sqrt( std::numeric_limits<double>::epsilon() );
184 
185  // 3D/face case.
186  if ( 2 == ref_dim )
187  {
188  DTK_CHECK( 3 == reference_point.size() );
189  DTK_CHECK( 3 == normal.size() );
190 
191  // Create extra points.
192  Teuchos::Array<double> ref_p1( reference_point );
193  Teuchos::Array<double> ref_p2( reference_point );
194 
195  // Apply a perturbation to the extra points.
196  double p1_sign = 1.0;
197  ref_p1[0] += perturbation;
198  if ( !this->checkPointInclusion(entity,ref_p1()) )
199  {
200  ref_p1[0] -= 2*perturbation;
201  p1_sign = -1.0;
202  }
203  double p2_sign = 1.0;
204  ref_p2[1] += perturbation;
205  if ( !this->checkPointInclusion(entity,ref_p2()) )
206  {
207  ref_p2[1] -= 2*perturbation;
208  p2_sign = -1.0;
209  }
210 
211  // Map the perturbed points to the physical frame.
212  Teuchos::Array<double> p0( physical_dim );
213  this->mapToPhysicalFrame( entity, reference_point(), p0() );
214  Teuchos::Array<double> p1( physical_dim );
215  this->mapToPhysicalFrame( entity, ref_p1(), p1() );
216  Teuchos::Array<double> p2( physical_dim );
217  this->mapToPhysicalFrame( entity, ref_p2(), p2() );
218 
219  // Compute the cross product of the tangents produced by the
220  // perturbation.
221  Teuchos::Array<double> tan1( physical_dim );
222  Teuchos::Array<double> tan2( physical_dim );
223  for ( int d = 0; d < physical_dim; ++d )
224  {
225  tan1[d] = p1_sign*(p1[d] - p0[d]);
226  tan2[d] = p2_sign*(p2[d] - p0[d]);
227  }
228  normal[0] = tan1[1]*tan2[2] - tan1[2]*tan2[1];
229  normal[1] = tan1[2]*tan2[0] - tan1[0]*tan2[2];
230  normal[2] = tan1[0]*tan2[1] - tan1[1]*tan2[0];
231  }
232 
233  // 2D/edge case.
234  else if ( 1 == ref_dim )
235  {
236  DTK_CHECK( 2 == reference_point.size() );
237  DTK_CHECK( 2 == normal.size() );
238 
239  // Create extra points.
240  Teuchos::Array<double> ref_p1( reference_point );
241 
242  // Apply a perturbation to the extra points.
243  double p1_sign = 1.0;
244  ref_p1[0] += perturbation;
245  if ( !this->checkPointInclusion(entity,ref_p1()) )
246  {
247  ref_p1[0] -= 2*perturbation;
248  p1_sign = -1.0;
249  }
250 
251  // Map the perturbed points to the physical frame.
252  Teuchos::Array<double> p0( physical_dim );
253  this->mapToPhysicalFrame( entity, reference_point(), p0() );
254  Teuchos::Array<double> p1( physical_dim );
255  this->mapToPhysicalFrame( entity, ref_p1(), p1() );
256 
257  // Compute the cross product of the tangents produced by the
258  // perturbation.
259  Teuchos::Array<double> tan( physical_dim );
260  for ( int d = 0; d < physical_dim; ++d )
261  {
262  tan[d] = p1_sign*(p1[d] - p0[d]);
263  }
264  normal[0] = -tan[0];
265  normal[1] = tan[1];
266  }
267 
268  // Normalize the normal vector.
269  double norm = 0.0;
270  for ( int d = 0; d < physical_dim; ++d )
271  {
272  norm += normal[d]*normal[d];
273  }
274  norm = std::sqrt(norm);
275  for ( int d = 0; d < physical_dim; ++d )
276  {
277  normal[d] /= norm;
278  }
279 }
