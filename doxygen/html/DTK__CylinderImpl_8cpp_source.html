206 bool CylinderImpl::checkPointInclusion(
207  const double tolerance,
208  const Teuchos::ArrayView<const double>& reference_point ) const
209 {
210  DTK_REQUIRE( reference_point.size() == 3 );
211 
212  double x_dist = d_centroid_x - reference_point[0];
213  double y_dist = d_centroid_y - reference_point[1];
214  double r = std::sqrt( x_dist*x_dist + y_dist*y_dist );
215  double rad_tol = d_radius*tolerance;
216  double half_length_tol = d_length / 2 + d_length*tolerance;
217 
218  return ( (r <= d_radius + rad_tol) &&
219  (reference_point[2] >= d_centroid_z - half_length_tol) &&
220  (reference_point[2] <= d_centroid_z + half_length_tol) );
221 }
